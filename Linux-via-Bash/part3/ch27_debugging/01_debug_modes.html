От основ Linux к мастерству Bash

Режимы отладки

set -x (xtrace)

Выводит команды перед выполнением.

Использование

#!/bin/bash
set -x

command1
command2

set +x  # Отключить

Пример вывода

$ ./script.sh
+ command1
output1
+ command2
output2

Вывод в файл

#!/bin/bash

# В отдельный FD
exec 19>debug.log
BASH_XTRACEFD=19
set -x

command1
command2

set +x
exec 19>&-

set -v (verbose)

Выводит строки кода перед выполнением.

#!/bin/bash
set -v

command1  # Выведет саму строку
command2

set -n (noexec)

Не выполняет команды, только проверка синтаксиса.

#!/bin/bash
set -n

# Эти команды не выполнятся
command1
command2

Комбинирование

# Полная отладка
set -xv

# Только вывод команд
set -x

# Только вывод строк
set -v

Отладка при запуске

# С флагом -x
bash -x script.sh

# С флагом -v
bash -v script.sh

# С флагом -n (проверка)
bash -n script.sh

Частичная отладка

#!/bin/bash

# Без отладки
command1

# Включить отладку
set -x
debug_command1
debug_command2
set +x

# Снова без отладки
command2

Отладка функций

#!/bin/bash

myfunc() {
    set -x
    # Тело функции
    set +x
}

# Или глобально
set -x
myfunc  # Функция тоже трассируется

Практические примеры

Логирование с временем

#!/bin/bash

export PS4='+ $(date +%T) ${BASH_SOURCE}:${LINENO}: '
set -x

command1
command2

Условная отладка

#!/bin/bash

if [[ $DEBUG = 1 ]]; then
    set -x
fi

command1
command2

Отладка с функцией

#!/bin/bash

debug_on() {
    exec 19>/tmp/debug.log
    BASH_XTRACEFD=19
    set -x
}

debug_off() {
    set +x
    exec 19>&-
}

debug_on
command1
debug_off

PS4 — формат вывода

# По умолчанию
PS4='+ '

# С номером строки
PS4='+ ${BASH_SOURCE}:${LINENO}: '

# С функцией и строкой
PS4='+ ${FUNCNAME[0]:-${BASH_SOURCE}:${LINENO}}: '

# С временем
PS4='+ $(date +%T.%N) ${BASH_SOURCE}:${LINENO}: '

Распространённые ошибки

Забыли отключить

# НЕПРАВИЛЬНО:
set -x
command1
# set +x забыли!

# ПРАВИЛЬНО:
set -x
command1
set +x

Вывод в stderr

# Отладка смешивается с выводом
./script.sh 2>&1 | grep error

# Разделить
./script.sh 2>/dev/null | grep error
