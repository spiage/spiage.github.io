select — создание меню

select создаёт интерактивное меню выбора.

Синтаксис

select name [in words ...]; do
    commands
done

Описание

  1. Разворачивает words в список элементов
  2. Выводит пронумерованный список на stderr
  3. Показывает приглашение PS3
  4. Читает ввод пользователя
  5. Если введён номер — устанавливает name в соответствующее слово
  6. Пустая строка — показывает список снова
  7. EOF — завершает с статусом 1
  8. Выполняет commands до break

Переменные

ПеременнаяОписание
nameВыбранное слово
REPLYВведённая строка
PS3Приглашение (по умолчанию: #? )

Примеры

Простое меню

select fruit in apple banana cherry; do
    echo "Вы выбрали: $fruit"
    break
done

Вывод:

1) apple
2) banana
3) cherry
#? 2
Вы выбрали: banana

Меню с PS3

PS3="Выберите фрукт: "
select fruit in apple banana cherry; do
    echo "Ваш выбор: $fruit (номер: $REPLY)"
    break
done

Меню с проверкой

PS3="Выберите действие: "
select action in "Создать" "Удалить" "Выход"; do
    case $REPLY in
        1) echo "Создаю...";;
        2) echo "Удаляю...";;
        3) echo "Выход"; exit 0;;
        *) echo "Неверный выбор";;
    esac
done

Меню с позиционными параметрами

# Без in words — использует $@
select arg; do
    echo "Выбрано: $arg"
    break
done
# Аргументы скрипта становятся элементами меню

Циклическое меню

PS3="Действие: "
while true; do
    select action in "Добавить" "Удалить" "Выход"; do
        case $action in
            "Добавить")
                echo "Добавляю..."
                break
                ;;
            "Удалить")
                echo "Удаляю..."
                break
                ;;
            "Выход")
                echo "Выход"
                exit 0
                ;;
            *)
                echo "Неверный выбор"
                ;;
        esac
    done
done

Меню с массивом

options=("Первый" "Второй" "Третий")
PS3="Выберите: "
select opt in "${options[@]}"; do
    echo "Выбрано: $opt"
    break
done

Меню с файлами

PS3="Выберите файл: "
select file in *.txt; do
    if [[ -n $file ]]; then
        echo "Открываю: $file"
        cat "$file"
        break
    else
        echo "Неверный выбор"
    fi
done

Многоуровневое меню

while true; do
    PS3="Главное меню: "
    select main in "Файл" "Правка" "Выход"; do
        case $main in
            "Файл")
                PS3="Файл: "
                select file in "Новый" "Открыть" "Сохранить"; do
                    echo "Файл -> $file"
                    break
                done
                break
                ;;
            "Правка")
                echo "Правка..."
                break
                ;;
            "Выход")
                exit 0
                ;;
            *)
                echo "Неверный выбор"
                ;;
        esac
    done
done

Ввод с проверкой

PS3="Выберите (1-3): "
select opt in one two three; do
    if [[ -n $opt ]]; then
        echo "Выбрано: $opt"
        break
    else
        echo "Неверный номер"
    fi
done

Специальные возможности

Обработка пустого ввода

PS3="Выбор: "
select opt in a b c; do
    if [[ -z $opt ]]; then
        echo "Пустой ввод, попробуйте снова"
    else
        echo "Выбрано: $opt"
        break
    fi
done

Обработка EOF

select opt in a b c; do
    echo "Выбрано: $opt"
    break
done || echo "Ввод завершён (EOF)"

Кастомное форматирование

# Сохраняем оригинальный PS3
old_ps3=$PS3

# Настраиваем своё приглашение
PS3=$'\n''Ваш выбор > '

select opt in one two three; do
    echo "OK: $opt"
    break
done

# Восстанавливаем
PS3=$old_ps3

Практические примеры

Конфигуратор

#!/bin/bash

PS3="Настройка: "
select option in "Язык" "Порт" "Выход"; do
    case $option in
        "Язык")
            select lang in "Русский" "English"; do
                echo "Язык: $lang"
                break
            done
            ;;
        "Порт")
            read -p "Порт: " port
            echo "Порт: $port"
            ;;
        "Выход")
            exit 0
            ;;
        *)
            echo "Неверный выбор"
            ;;
    esac
done

Установщик

#!/bin/bash

PS3="Компонент для установки: "
select component in "Ядро" "Драйверы" "Утилиты" "Готово"; do
    case $component in
        "Ядро")
            echo "Устанавливаю ядро..."
            ;;
        "Драйверы")
            echo "Устанавливаю драйверы..."
            ;;
        "Утилиты")
            echo "Устанавливаю утилиты..."
            ;;
        "Готово")
            echo "Установка завершена"
            break
            ;;
        *)
            echo "Неверный выбор"
            ;;
    esac
done

Отладочное меню

if [[ $DEBUG = 1 ]]; then
    PS3="Отладка: "
    select action in "Дамп" "Трассировка" "Продолжить"; do
        case $action in
            "Дамп") dump_state;;
            "Трассировка") set -x;;
            "Продолжить") break;;
        esac
    done
fi

Статус выхода

  • 0 — выполнен break
  • 1 — прочитан EOF
  • !=0 — статус последней команды

Распространённые ошибки

Отсутствие break

# НЕПРАВИЛЬНО (бесконечный цикл):
select opt in a b c; do
    echo "Выбрано: $opt"
    # Нет break!
done

# ПРАВИЛЬНО:
select opt in a b c; do
    echo "Выбрано: $opt"
    break
done

Неверная проверка ввода

# НЕПРАВИЛЬНО:
select opt in a b c; do
    if [[ $opt = "" ]]; then  # Не работает!
        echo "Пусто"
    fi
done

# ПРАВИЛЬНО:
select opt in a b c; do
    if [[ -z $opt ]]; then
        echo "Пусто"
    fi
done
