Дополнительные утилиты

Набор инструментов для диагностики и мониторинга системы.

ps — информация о процессах

Базовое использование

# Все процессы
ps aux

# Вывод:
# USER   PID %CPU %MEM    VSZ   RSS TTY      STAT START   TIME COMMAND
# root     1  0.0  0.1 169936 11776 ?        Ss   Mar13   0:02 /sbin/init
# user  1234  0.1  0.5 500000 50000 pts/0    S+   10:00   0:01 bash

Колонки:

  • STAT — состояние (R, S, D, Z, T)
  • VSZ — виртуальная память
  • RSS — резидентная память
  • %CPU, %MEM — использование ресурсов

Фильтрация

# Процессы пользователя
ps -u username

# По имени
ps -C bash

# По PID
ps -p 1234,5678

# Дерево процессов
ps auxf

Форматирование

# Custom вывод
ps -eo pid,ppid,user,%cpu,%mem,comm

# Сортировка по CPU
ps aux --sort=-%cpu | head -10

# Сортировка по памяти
ps aux --sort=-%mem | head -10

vmstat — статистика виртуальной памяти

# Обновление каждую секунду
vmstat 1

# Вывод:
# procs -----------memory---------- ---swap-- -----io---- -system-- ------cpu-----
#  r  b   swpd   free   buff  cache   si   so    bi    bo   in   cs us sy id wa st
#  1  0      0 2048000 256000 4096000    0    0    10    20  100  200 10  2 88  0  0

Колонки:

  • r — процессы в очереди на CPU
  • b — процессы в sleep
  • swpd — использовано swap
  • free — свободная память
  • bi/bo — блоки в/из
  • us/sy/id/wa — CPU время (user/system/idle/wait)

iostat — статистика I/O

# Обновление каждую секунду
iostat -x 1

# Вывод:
# avg-cpu:  %user   %nice %system %iowait  %steal   %idle
#          10.00    0.00    2.00    5.00    0.00   83.00

# Device            r/s     w/s     rkB/s     wkB/s   %util
# sda              10.00   20.00    100.00    200.00   15.00

Практические примеры

Мониторинг скрипта

#!/bin/bash

PID=$$

# Логировать ресурсы
while kill -0 $PID 2>/dev/null; do
    ps -p $PID -o pid,%cpu,%mem,vsz,rss >> /var/log/script_resources.log
    sleep 60
done

Поиск проблем с памятью

#!/bin/bash

# Процессы с большим потреблением памяти
ps aux --sort=-%mem | head -10

# Проверить swap
vmstat 1 | grep -v "procs" | awk '$3 > 0 {print "Swap in progress!"}'

Диагностика медленной системы

#!/bin/bash

echo "=== CPU Load ==="
cat /proc/loadavg

echo -e "\n=== Memory ==="
free -h

echo -e "\n=== Top Processes by CPU ==="
ps aux --sort=-%cpu | head -5

echo -e "\n=== Top Processes by Memory ==="
ps aux --sort=-%mem | head -5

echo -e "\n=== Disk I/O ==="
iostat -x 1 2 | tail -20

Проверка доступности порта

#!/bin/bash

PORT=8080

if ss -tlnp | grep -q ":$PORT "; then
    echo "Port $PORT is in use"
    ss -tlnp | grep ":$PORT "
else
    echo "Port $PORT is free"
fi

Автоматический restart при утечке памяти

#!/bin/bash

MAX_RSS=1000000  # kB
PIDFILE="/var/run/myapp.pid"

while true; do
    if [[ -f "$PIDFILE" ]]; then
        PID=$(cat "$PIDFILE")
        RSS=$(grep VmRSS /proc/$PID/status 2>/dev/null | awk '{print $2}')
        
        if [[ $RSS -gt $MAX_RSS ]]; then
            echo "Memory limit exceeded, restarting..."
            kill $PID
            sleep 2
            start_myapp
        fi
    fi
    
    sleep 60
done
