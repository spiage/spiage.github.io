От основ Linux к мастерству Bash

setuid, setgid, sticky

Специальные биты для изменения поведения прав доступа.

setuid (Set User ID)

Что делает

setuid на файле заставляет процесс выполняться от владельца файла, а не от запустившего пользователя.

# Установить setuid
chmod u+s /usr/bin/program

# Или числом:
chmod 4755 /usr/bin/program
# ↑ 4 = setuid

Пример: passwd

# passwd должна менять /etc/shadow
# Но /etc/shadow доступен только root

ls -l /usr/bin/passwd
# -rwsr-xr-x 1 root root ... /usr/bin/passwd
#   ↑ 's' = setuid

# При запуске:
# 1. Процесс запускается от обычного пользователя
# 2. Но получает Effective UID = 0 (root)
# 3. Может менять /etc/shadow

Просмотр

# Найти файлы с setuid
find / -perm -4000 2>/dev/null

# Вывод:
# /usr/bin/passwd
# /usr/bin/sudo
# /usr/bin/su

Опасности

# setuid + скрипт = ОПАСНО!
# Скрипты с setuid игнорируются ядром

# setuid + уязвимость = компрометация
# Если в программе с setuid есть баг — злоумышленник получает root

setgid (Set Group ID)

Что делает

setgid на файле заставляет процесс выполняться от группы файла.

# Установить setgid
chmod g+s /usr/bin/program

# Или числом:
chmod 2755 /usr/bin/program
# ↑ 2 = setgid

setgid на директории

Полезное применение: файлы в директории наследуют группу:

# Создать общую директорию
mkdir /shared
chown root:developers /shared
chmod 2775 /shared
# ↑ setgid на директории

# Теперь все файлы в /shared будут группы developers
touch /shared/file.txt
ls -l /shared/file.txt
# -rw-rw-r-- 1 user developers ... /shared/file.txt

Просмотр

# Найти файлы с setgid
find / -perm -2000 2>/dev/null

# Найти директории с setgid
find / -type d -perm -2000 2>/dev/null

sticky bit

Что делает

sticky на директории запрещает удаление файлов кроме:

  • Владельца файла
  • Владельца директории
  • root
# Установить sticky
chmod +t /tmp

# Или числом:
chmod 1777 /tmp
# ↑ 1 = sticky

Пример: /tmp

ls -ld /tmp
# drwxrwxrwt 10 root root ... /tmp
#                 ↑ 't' = sticky

# Пользователь user1 может создать файл
touch /tmp/file1

# Пользователь user2 НЕ может удалить чужой файл
rm /tmp/file1
# rm: remove write-protected regular empty file 'file1'?
# (даже при правах на запись в /tmp)

Просмотр

# Найти директории с sticky
find / -type d -perm -1000 2>/dev/null

# Вывод:
# /tmp
# /var/tmp

Практические примеры

Общая директория для проекта

#!/bin/bash

# Создать общую директорию
PROJECT_DIR="/opt/project"
GROUP="developers"

mkdir -p "$PROJECT_DIR"
chown root:"$GROUP" "$PROJECT_DIR"
chmod 2775 "$PROJECT_DIR"

echo "Shared directory created"
echo "Files will inherit group: $GROUP"

Безопасный временный каталог

#!/bin/bash

# Создать безопасный временный каталог
SAFE_TMP=$(mktemp -d)
chmod 1700 "$SAFE_TMP"

# Только владелец может читать/писать
# Остальные не видят содержимое

echo "Safe temp: $SAFE_TMP"

# Очистка
trap 'rm -rf "$SAFE_TMP"' EXIT

Проверка setuid программ

#!/bin/bash

# Аудит setuid файлов
echo "=== SUID Files ==="
find / -perm -4000 -type f 2>/dev/null | while read file; do
    echo "$file"
    # Проверить пакет
    dpkg -S "$file" 2>/dev/null || echo "  (unknown package)"
done

echo -e "\n=== SGID Files ==="
find / -perm -2000 -type f 2>/dev/null

Удаление setuid

#!/bin/bash

# Удалить setuid с файла
chmod u-s /usr/bin/program

# Удалить setgid
chmod g-s /usr/bin/program

# Удалить sticky
chmod -t /tmp

# Удалить все специальные биты
chmod a-s /usr/bin/program

Безопасность

Риски setuid

  1. Уязвимости — баг в setuid программе = компрометация
  2. Скрипты — setuid на скриптах не работает (игнорируется)
  3. Интерпретаторы — setuid на скриптах bash/python игнорируется

Аудит

#!/bin/bash

# Найти все setuid/setgid
find / \( -perm -4000 -o -perm -2000 \) -type f 2>/dev/null

# Проверить неизвестные файлы
find / -perm -4000 -type f 2>/dev/null | while read f; do
    if ! dpkg -S "$f" >/dev/null 2>&1; then
        echo "Unknown SUID: $f"
    fi
done

Удаление лишних setuid

#!/bin/bash

# Удалить setuid с опасных программ
chmod u-s /usr/bin/chsh
chmod u-s /usr/bin/chfn
chmod u-s /usr/bin/newgrp
chmod u-s /usr/bin/gpasswd

# Осторожно! Может сломать функциональность!
