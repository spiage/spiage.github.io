От основ Linux к мастерству Bash

Библиотечные обёртки над системными вызовами

Программы не вызывают ядро напрямую — они используют библиотеки.

Разница между syscall и библиотечной функцией

Приложение
    ↓
Библиотечная функция (libc)
    ↓
Системный вызов (ядро)
    ↓
Оборудование

Пример: printf vs write

// Библиотечная функция
printf("Hello, World!\n");

// Внутри printf:
// 1. Форматирование строки
// 2. Буферизация
// 3. write(1, "Hello, World!\n", 14)

// Прямой syscall (через syscall())
syscall(SYS_write, 1, "Hello, World!\n", 14);

В bash:

# printf — builtin, использует write
strace -e trace=write bash -c 'printf "hello"'
# write(1, "hello", 5) = 5

# /usr/bin/printf — внешняя команда
strace -e trace=write /usr/bin/printf "hello"
# + много других syscall для инициализации

Почему нужны обёртки

1. Переносимость

Системные вызовы различаются между архитектурами:

// Вместо прямого syscall (зависит от архитектуры)
long ret = syscall(SYS_write, fd, buf, count);

// Используем переносимую функцию
ssize_t ret = write(fd, buf, count);

Пример для разных архитектур:

x86_64:  SYS_write = 1
ARM64:   SYS_write = 64
RISC-V:  SYS_write = 64

В bash:

# bash сам определяет архитектуру
uname -m
# x86_64

# Но write() работает одинаково
echo "hello" > file.txt

2. Буферизация

stdio функции буферизируют вывод для производительности:

// printf буферизирует
for (int i = 0; i < 1000; i++) {
    printf("Line %d\n", i);  // В буфере
}
fflush(stdout);  // Теперь записано (один write вместо 1000)

// Без буферизации
for (int i = 0; i < 1000; i++) {
    syscall(SYS_write, 1, buf, len);  // 1000 syscall!
}

В bash:

# Встроенный echo — буферизация в ядре
time for i in {1..1000}; do echo test > /dev/null; done

# Внешний echo — fork + exec overhead
time for i in {1..1000}; do /bin/echo test > /dev/null; done

3. Обработка ошибок

Libc преобразует коды ошибок ядра в errno:

// Ядро возвращает -EPERM
int fd = open("/root/file", O_RDONLY);
if (fd == -1) {
    // errno установлен в EPERM
    perror("open");  // "open: Operation not permitted"
}

// Прямой syscall
long ret = syscall(SYS_open, "/root/file", O_RDONLY);
if (ret == -1) {
    // errno тоже установлен
    perror("syscall");
}

В bash:

# $? содержит код возврата
cat /etc/shadow 2>/dev/null
echo $?  # 1 (ошибка)

# strace показывает errno ядра
strace cat /etc/shadow 2>&1 | grep EACCES
# openat(..., "/etc/shadow", ...) = -1 EACCES

4. Дополнительные возможности

// fopen делает больше чем open:
FILE *f = fopen("file.txt", "r");
// 1. Вызывает open()
// 2. Выделяет память для FILE*
// 3. Настраивает буферизацию
// 4. Возвращает поток

// vs прямой open:
int fd = open("file.txt", O_RDONLY);
// Только FD, нет буферизации

В bash:

# read builtin — буферизированный
while read line; do
    echo "$line"
done < file.txt

# Прямой read из /dev/stdin — без буферизации
dd if=/dev/stdin bs=1 count=10

Как увидеть разницу

strace

# printf из bash (builtin) — один write
strace -e write bash -c 'printf "hello"'
# write(1, "hello", 5) = 5

# /usr/bin/printf — много syscall
strace -e write /usr/bin/printf "hello"
# + mmap, open, read для инициализации libc
# write(1, "hello", 5) = 5

ldd

# Какие библиотеки использует команда
ldd /bin/ls
# linux-vdso.so.1
# libc.so.6 => /lib/x86_64-linux-gnu/libc.so.6
# libselinux.so.1 => ...

# builtin команды не имеют зависимостей
type cd
# cd is a shell builtin

/proc

# Библиотеки процесса
cat /proc/$$/maps | grep "\.so"

# Или
ldd /proc/$$/exe

Влияние на скрипты bash

Встроенные команды быстрее

#!/bin/bash

# Встроенные (нет fork/exec)
time for i in {1..10000}; do
    :  # builtin, ничего не делает
done

# Внешние (fork + exec)
time for i in {1..10000}; do
    true  # /bin/true
done

Избегайте лишних внешних команд

# ПЛОХО: внешняя команда в цикле
for file in *.txt; do
    dirname "$file"    # fork для каждого файла
done

# ХОРОШО: builtin или parameter expansion
for file in *.txt; do
    dir="${file%/*}"   # builtin, нет fork
done

Конвейеры и буферизация

# printf буферизирует
printf "%s\n" {1..1000} | cat

# vs построчный вывод
for i in {1..1000}; do
    echo "$i"  # Каждый echo — write()
done | cat

Когда обёртки скрывают проблемы

1. Буферизация маскирует проблемы

// printf может успешно завершиться, но данные не записаны
printf("data");  // В буфере libc
// Программа упала до fflush() — данные потеряны

// write гарантирует запись в ядро (но не на диск!)
write(fd, "data", 4);  // В ядре

В bash:

# Данные могут быть в буфере ядра
echo "critical" > file.txt
# sync  # Сбросить на диск

# Или использовать dd с sync
dd if=/dev/stdin of=file.txt conv=fsync

2. errno может быть перезаписан

// Ошибка в одной функции
open("/nonexistent", O_RDONLY);  // errno = ENOENT

// Другая функция перезаписывает errno
close(0);  // errno может измениться!

// Нужно сохранить сразу
if (open("/nonexistent", O_RDONLY) == -1) {
    int saved_errno = errno;
    // ...
    errno = saved_errno;
}

В bash:

# $? сбрасывается после каждой команды
false
echo $?  # 1

echo "test"
echo $?  # 0 (предыдущий код потерян!)

# Сохранить сразу
false
rc=$?
echo "Exit code: $rc"

3. setvbuf может изменить поведение

// Изменить буферизацию stdout
setvbuf(stdout, NULL, _IONBF, 0);  // Без буфера

// Теперь printf пишет сразу
printf("immediate\n");  // write() сразу

В bash:

# Включить unbuffered вывод
stdbuf -oL tail -f /var/log/syslog

# Или через script
script -q -c "tail -f /var/log/syslog" /dev/null

Практические примеры

Проверка буферизации

#!/bin/bash

# Буферизированный вывод
echo "Buffered" | cat

# Построчный (line buffered)
stdbuf -oL echo "Line buffered" | cat

# Без буфера
stdbuf -o0 echo "Unbuffered" | cat

Отладка через strace

#!/bin/bash

# Что делает команда на самом деле?
strace -e trace=read,write,openat command

# Сколько syscall?
strace -c command

# Какие библиотеки загружает?
strace -e trace=openat command 2>&1 | grep "\.so"

Оптимизация скрипта

#!/bin/bash

# ПЛОХО: много внешних команд
cat file.txt | grep "pattern" | cut -d: -f1 | sort | uniq

# ХОРОШО: меньше процессов
grep "pattern" file.txt | cut -d: -f1 | sort -u

# ЛУЧШЕ: awk вместо цепочки
awk -F: '/pattern/ {print $1}' file.txt | sort -u
