Сокеты

Сокеты — механизм IPC для локальной и сетевой связи.

Типы сокетов

ТипАдресное пространствоПример
Unix domainФайловая система/var/run/docker.sock
TCP/IPСеть (IP:port)127.0.0.1:8080
UDPСеть (IP:port)0.0.0.0:53

Unix domain сокеты

Создание

# Посмотреть сокеты
ss -xl  # Unix domain
ss -tl  # TCP

# Пример вывода:
# u_str LISTEN 0 128 /var/run/docker.sock 12345 * 0

Подключение

# netcat для Unix сокета
nc -U /var/run/docker.sock

# Отправить запрос
echo "GET /info HTTP/1.0\r\n\r\n" | nc -U /var/run/docker.sock

Пример: клиент-сервер

#!/bin/bash

SOCKET="/tmp/myapp.sock"

# Сервер
start_server() {
    rm -f "$SOCKET"
    socat UNIX-LISTEN:"$SOCKET",fork EXEC:"handle_request.sh"
}

# Обработчик
handle_request() {
    read request
    echo "Response to: $request"
}

# Клиент
send_request() {
    echo "Hello" | socat - UNIX-CONNECT:"$SOCKET"
}

TCP/IP сокеты

Прослушивание

# Простой сервер на netcat
nc -l 8080

# С обработкой подключений
while true; do
    nc -l 8080 | handle_request
done

Подключение

# Подключиться к серверу
nc localhost 8080

# С таймаутом
nc -w 5 localhost 8080

# Отправить данные
echo "request" | nc localhost 8080

Сервер с обработкой

#!/bin/bash

PORT=8080

# Слушать и обрабатывать
while true; do
    nc -l $PORT | \
    while read request; do
        echo "Processing: $request"
        echo "Response" | nc -q1 localhost $PORT
    done
done

socat

socat — универсальный инструмент для сокетов.

Примеры

# TCP сервер → программа
socat TCP-LISTEN:8080,fork EXEC:"./handler.sh"

# Unix сокет → TCP
socat UNIX-LISTEN:/tmp/app.sock,fork TCP:remote:8080

# TCP → файл (логирование)
socat TCP-LISTEN:8080,fork OPEN:/var/log/requests.log,append

Двусторонняя связь

#!/bin/bash

# Сервер
socat TCP-LISTEN:9000,fork SYSTEM:"bash"

# Клиент
socat TCP:localhost:9000 EXEC:"bash"

Практические примеры

HTTP сервер

#!/bin/bash

PORT=8080

handle_request() {
    read request
    echo "HTTP/1.1 200 OK"
    echo "Content-Type: text/plain"
    echo ""
    echo "Hello from bash!"
}

export -f handle_request

socat TCP-LISTEN:$PORT,fork SYSTEM:"handle_request"

Прокси

#!/bin/bash

# Прокси с localhost:8080 на remote:3306
socat TCP-LISTEN:8080,fork TCP:remote-host:3306

Мониторинг сокетов

#!/bin/bash

# Проверить кто слушает порт
ss -tlnp | grep :8080

# Проверить подключения
ss -tnp | grep :8080

# Статистика
netstat -s | grep -i tcp

Балансировка

#!/bin/bash

# Простой round-robin балансировщик
BACKENDS=("host1:8080" "host2:8080" "host3:8080")
INDEX=0

while true; do
    backend=${BACKENDS[$INDEX]}
    socat TCP-LISTEN:80,fork TCP:$backend &
    INDEX=$(( (INDEX + 1) % ${#BACKENDS[@]} ))
done

Сокеты в скриптах

/dev/tcp (bash extension)

# Отправить HTTP запрос
exec 3<>/dev/tcp/www.google.com/80
echo -e "GET / HTTP/1.0\r\n" >&3
cat <&3
exec 3>&-

Проверка доступности

#!/bin/bash

check_port() {
    local host=$1
    local port=$2
    
    timeout 5 bash -c "echo > /dev/tcp/$host/$port" 2>/dev/null
    return $?
}

if check_port localhost 3306; then
    echo "MySQL is up"
else
    echo "MySQL is down"
fi

Простой клиент

#!/bin/bash

HOST="localhost"
PORT="8080"

send_request() {
    exec 3<>/dev/tcp/$HOST/$PORT
    echo "$1" >&3
    read response <&3
    exec 3>&-
    echo "$response"
}

result=$(send_request "Hello")
echo "Server responded: $result"

Безопасность

Ограничение доступа к Unix сокету

# Создать с правами
socat UNIX-LISTEN:/tmp/app.sock,mode=0660,fork ...

# Изменить владельца
chown user:group /tmp/app.sock

Firewall для TCP

# Разрешить только localhost
iptables -A INPUT -p tcp --dport 8080 -s 127.0.0.1 -j ACCEPT
iptables -A INPUT -p tcp --dport 8080 -j DROP

TLS шифрование

# Сервер с TLS
socat OPENSSL-LISTEN:443,cert=server.pem,verify=0,fork ...

# Клиент с TLS
socat OPENSSL:server.com:443,verify=0 ...

Практические рекомендации

  1. Используйте Unix сокеты для локальной связи (быстрее)
  2. Ограничивайте права доступа к сокетам
  3. Обрабатывайте таймауты при сетевой связи
  4. Логируйте подключения для отладки
